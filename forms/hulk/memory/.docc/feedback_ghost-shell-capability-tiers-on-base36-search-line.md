public struct GhostCapabilityTier: Hashable, Sendable, Comparable, CustomStringConvertible {
  /// Position on the base36 search-line — the typed PI rating equivalent.
  public let triplet: GhostShellSearchTriplet

  /// 78 typed labels = 26 letter-classes (A through Z) × 3 sub-tiers per
  /// class. Constructed from (letterClass, withinClassRank).

    case runnerUp   = 1  // double-letter label (e.g. "AA") — second rank
    case best       = 2  // triple-letter label (e.g. "AAA") — best in class

  /// Letter-class A..Z (operator-legible alphabet position).
  public let letterClass: Character  // "A" through "Z"

  /// Within-class rank (single/double/triple letter repetition).
  public let withinClassRank: WithinClassRank

  /// Render as the operator-legible label, e.g. "A", "AA", "AAA", "B", ...
  public var description: String {
    String(repeating: letterClass, count: withinClassRank.rawValue + 1)
  }

  /// Lexicographic ordering matches the substrate's typed capability
  /// ordering. A < AA < AAA < B < BB < BBB < ... < Z < ZZ < ZZZ.
  public static func < (lhs: Self, rhs: Self) -> Bool {
    if lhs.letterClass != rhs.letterClass { return lhs.letterClass < rhs.letterClass }
    return lhs.withinClassRank < rhs.withinClassRank
  }
}
